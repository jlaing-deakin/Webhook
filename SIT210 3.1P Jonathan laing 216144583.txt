// This #include statement was automatically added by the Particle IDE.#include <HC-SR04.h>double distance = 0.0;int trigPin = D4;int echoPin = D5;HC_SR04 rangefinder = HC_SR04(trigPin, echoPin);void setup(){    Serial.begin(9600);    pinMode(D7, OUTPUT);    //Particle.variable("distance", &distance, DOUBLE);}void loop(){    distance = rangefinder.distInch();        Particle.publish("Distance is", distance);    blinkLed();    delay(5000);    }void blinkLed() {    digitalWrite(D7,HIGH);    delay(150);       digitalWrite(D7,LOW);    }
